Scale values

for (col in colnames(setA)) {
    min_c <- min(setA[, col])
    max_c <- max(setA[, col])
    setA[, col] <- (setA[, col] - min_c) / (max_c - min_c)
}

sbTot <- c(length(A_labs[A_labs == "s"]), length(A_labs[A_labs == "b"]))
